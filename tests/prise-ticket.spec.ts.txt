import { test, expect } from '@playwright/test';

test("Le visiteur peut prendre un ticket avec succès", async ({ page }) => {
  // 1. Le robot va sur la page d'accueil (vérifie que c'est le bon port local, ex: 3000)
  await page.goto('http://localhost:3000/'); 

  // 2. Le robot cherche le champ pour le numéro de téléphone et tape un numéro de test
  // (Modifie le sélecteur 'input[type="tel"]' s'il est différent dans ton code)
  await page.fill('input[type="tel"]', '0612345678'); 

  // 3. Le robot clique sur le bouton "Prendre un ticket"
  // (Modifie le texte si ton bouton s'appelle autrement)
  await page.click('button:has-text("Prendre un ticket")');

  // 4. VERIFICATION : Le robot vérifie qu'un message de succès ou que le numéro du ticket apparaît
  // (Ici, il cherche le texte "Votre ticket". À adapter selon ce que ton app affiche vraiment)
  await expect(page.locator('text=Votre ticket')).toBeVisible();
});
